# 선택정렬

- `O(n^2)`의 정렬
- 작은 index부터 차례로 돌면서 정렬을 수행한다.

```python
def select_sort(list):
    length = len(list)
    for i in range(0, length-1):
        minIndex = i
        for j in range(i+1, length):
            if list[j] < list[minIndex]:
                minIndex = j

        list[i], list[minIndex] = list[minIndex], list[i]

    return list


array = [1, 5, 7, 435, 1243, 345, 456, 435,
         234, 654, 6758, 785, 23, 43425, 34, 2734, 237496, 98234, 8882348, 423]
result = sort(array)
print('result =>', result)
// result => [1, 5, 7, 23, 34, 234, 345, 423, 435, 435, 456, 654, 785, 1243, 2734, 6758, 43425, 98234, 237496, 8882348]
```
